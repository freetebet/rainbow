CREATE TABLE RAIN_Products_tbl
(
    prdNo VARCHAR2(20) PRIMARY KEY,

    prdSdc VARCHAR2(20) NOT NULL,

    prdMajorCtg VARCHAR2(50) NOT NULL,
    prdSubCtg VARCHAR2(100) NOT NULL,

    prdCstPri NUMBER, prdCstPri

    prdSelPrice NUMBER,
    prdSelSup NUMBER,
    prdSelTax NUMBER,

    prdSt VARCHAR2(20),
